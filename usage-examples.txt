SIMPLESED usage examples

// single occurence

simplesed input.txt DELETE "hello"
simplesed input.txt REPLACE "hello" WITH "some-text"
simplesed input.txt APPEND "hello" WITH "some-text"
simplesed input.txt PREPEND "hello" WITH "some-text"
simplesed input.txt WRAP "hello" WITH "some-text"


// lines

simplesed input.txt {DELETE/REPLACE/APPEND/PREPEND/WRAP} LINES CONTAINING 
"hello"
simplesed input.txt {DELETE/REPLACE/APPEND/PREPEND/WRAP} LINES STARTING WITH "hello"
simplesed input.txt {DELETE/REPLACE/APPEND/PREPEND/WRAP} LINES ENDING WITH "hello"

// advanced lines
simplesed input.txt {BEFORE/AFTER} LINES {CONTAINING/STARTING WITH/ENDING WITH} "hello" INSERT "This is a new line!"


// prefilters

simplesed input.txt IN LINES {WITH/STARTING WITH/ENDING WITH} world {DELETE/REPLACE/APPEND/PREPEND/WRAP} hello WITH world
