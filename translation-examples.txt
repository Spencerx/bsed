SIMPLE SED translation examples

// single occurrence

simplesed input.txt DELETE "hello"
sed ’s/hello//g’ input.txt

simplesed input.txt REPLACE "hello" WITH "some-text"
sed ’s/hello/some-text/‘ input.txt

simplesed input.txt APPEND "hello" WITH "some-text"
sed ’s/hello/hellosome-text/‘ input.txt

simplesed input.txt PREPEND "hello" WITH "some-text"
sed ’s/hello/some-texthello/‘ input.txt

simplesed input.txt WRAP "hello" WITH "some-text"
sed ’s/hello/some-texthellosome-text/‘ input.txt


// lines	

simplesed input.txt {DELETE/REPLACE/APPEND/PREPEND/WRAP} LINES CONTAINING 
"hello"

simplesed input.txt CLEAR LINES STARTING WITH hello
sed 's/^hello.*//g' input.txt
simplesed input.txt CLEAR LINES CONTAINING hello
sed 's/^.*hello.*//g'
simplesed input.txt CLEAR LINES ENDING WITH hello
sed 's/.*hello\n//g'

simplesed input.txt DELETE LINES STARTING WITH hello
sed '/^hello.*$/d' input.txt 


simplesed input.txt {DELETE/REPLACE/APPEND/PREPEND/WRAP} LINES STARTING WITH "hello"
sed 's/^hello.*/some-text/‘ input.txt

simplesed input.txt {DELETE/REPLACE/APPEND/PREPEND/WRAP} LINES ENDING WITH "hello"

// advanced lines
simplesed input.txt {BEFORE/AFTER} LINES {CONTAINING/STARTING WITH/ENDING WITH} "hello" INSERT "This is a new line!"


// prefilters

simplesed input.txt IN LINES {WITH/STARTING WITH/ENDING WITH} world {DELETE/REPLACE/APPEND/PREPEND/WRAP} hello WITH world


sed 's/hello/world/' input.txt > output.txt
simplesed input.txt REPLACE hello WITH world

sed 's/word-to-find//g' input.txt > output.file
simplesed input.txt DELETE word-to-find

simplesed input.txt APPEND word-to-find WITH some-text

simplesed input.txt PREPEND word-to-find WITH some-text

simplesed input.txt WRAP word-to-find WITH some-text


sed '/hello/d' input.txt > output.txt
simplesed input.txt DELETE LINES WITH hello
